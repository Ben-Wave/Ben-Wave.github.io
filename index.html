import requests
from bs4 import BeautifulSoup
import pandas as pd

def scrape_kleinanzeigen(search_term, max_pages=5):
    base_url = "https://www.ebay-kleinanzeigen.de/s-"
    results = []
    
    for page in range(1, max_pages+1):
        url = f"{base_url}{search_term}/seite:{page}"
        response = requests.get(url, headers={'User-Agent': 'Mozilla/5.0'})
        
        if response.status_code == 200:
            soup = BeautifulSoup(response.content, 'html.parser')
            listings = soup.find_all('article', class_='aditem')
            
            for item in listings:
                title = item.find('h2').text.strip()
                price = item.find('p', class_='aditem-main--middle--price').text.strip()
                location = item.find('div', class_='aditem-main--top--left').text.strip()
                
                results.append({
                    'title': title,
                    'price': price,
                    'location': location
                })
        
    return pd.DataFrame(results)

def create_file_manager(root_dir: str):

    # Breadcrumb navigation

    # File list

    def update_file_list():
        file_container.clear()
        with file_container:
            with ui.grid(columns=4).classes('gap-4'):
                for item in current_dir.iterdir():
                    if item.is_dir():
                        ui.card(item.name, on_click=lambda p=item: navigate_to(p)).classes('cursor-pointer')
                    else:
                        ui.card(f"{item.name}\n{item.stat().st_size} bytes").classes('bg-blue-50')
    
    def navigate_to(path):
        nonlocal current_dir
        current_dir = Path(path)
        update_breadcrumbs()
        update_file_list()
    
    update_breadcrumbs()
    update_file_list()

import ollama
from flask import Flask, request, jsonify
from langchain.memory import ConversationBufferMemory

app = Flask(__name__)
memory = ConversationBufferMemory(return_messages=True)

def format_prompt(user_input, context=""):
    return f"""
    {context}
    
    Benutzer: {user_input}
    Assistent: 
    """

@app.route('/chat', methods=['POST'])
def chat():
    data = request.json
    user_input = data.get('message', '')
    
    # Get conversation history
    history = memory.chat_memory.messages
    context = "\n".join([f"{msg.type.capitalize()}: {msg.content}" for msg in history[-6:]])
    
    # Generate response with Ollama
    response = ollama.generate(
        model='mistral',
        prompt=format_prompt(user_input, context),
        options={
            'temperature': 0.7,
            'top_p': 0.9,
            'max_tokens': 500
        }
    )
    
    # Save to memory
    memory.chat_memory.add_user_message(user_input)
    memory.chat_memory.add_ai_message(response['response'])
    
    return jsonify({"response": response['response']})
